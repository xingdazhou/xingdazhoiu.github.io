# myVuex.js

//1.维护状态state
//2.修改状态commit
//3。业务逻辑控制dispatch
//4.状态派发getter
//5.实现state响应式
//6.插件n
//7.混入

let myVue;

function install(_Vue,storeName = '$store'){
    myVue = _Vue;
    //混入 ；把store选项指定到vue原型上
    myVue.mixin({
        beforeCreate(){
            if(this.$options.store){
                myVue.prototype.$store = this.$options.store;//这个store是实例

            }
        }
    })

}

class Store{
    // 这个options就是平时配置的对象参数
    constructor(options = {}){
        // 这里利用vue数据响应式来管理state，将来state上面的值变了就会通知其他依赖更新
        this.state = new myVue({
            data:options.state
        })
        this.mutations = options.mutations || {};
        this.actions = options.actions || {};
        options.getters && this.handleGetters(options.getters);
    }
    // 要触发mutaions，需要实现commit方法
     commit = (name,...arg)=>{
        //  这里需要this指向store实例
        const fn = this.mutations[name];
        fn(this.state,...arg);
     }
     dispatch(name,...arg){
         const fn = this.actions[name];
         return fn(this,...arg)    //这里加return主要是可能actions里面函数执行返回promise然后dispatch接着返回
     }
    //  getters:{score(state){return xxx}}
     handleGetters(getters){
        this.getters = {};    //store实例上的getters
        // 定义属性只读
        Object.keys(getters).forEach(key =>{
            Object.defineProperty(this.getters,key,{   //这里的this.getters 一开始是 {}
                get:() => {
                    return getters[key](this.state)
                }
            })   
        })

     }

}

// 这里的导出对象就是Vuex
export default {Store,install}

# myVuexUse.js

import Vue from 'vue'
import Vuex from './myVuex'

Vue.use(Vuex)

export default new Vuex.Store({
    state:{
        count:0
    },
    mutations:{
        add(state,num){
            state.count += num
        }
    },
    getters:{
        score(state){
            return state.count
        }
    },
    actions:{
        addSync({commit},num){
            setTimeout(()=>{
                commit('add',num)
            },1000)
        }
    }
})

  name: 'App',

  },
  methods:{
    add(){
      this.$store.commit('add',1)   //做到了响应式，这里修改了state,通知了视图变化
