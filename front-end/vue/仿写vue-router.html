    # 模仿实现一个vue-router

    # myrouter.js

    // import vue from 'vue'
// 这里需要引入vue的响应式来管理当前current路径，但是这样引入又会给router插件增加负担
// 这里插件想用Vue就可以通过混入的时候传递进来
let Vue

// 定义router
class VueRouter {
  constructor(options) {
    this.$options = options
    // 创建一个路由path和route映射的对象
    this.routeMap = {}

    // 将来当前路径 current需要响应式
    // 借用vue响应式可以做到
    this.app = new Vue({
      data: {
        current: '/' //现在这个current是响应式的，地址栏hash值变了，所用用到current的地方也会变
      }
    })

  }

  init() {
    //   绑定浏览器事件
    this.bindEvent();
    //解析路由配置
    this.createRouteMap(this.$options)
    // 创建router-link和router-view
    this.initComponent();

  }

  bindEvent() {
    window.addEventListener("hashchange", this.onHashChange.bind(this)) //绑定当前this为组件实例
    window.addEventListener("load", this.onHashChange.bind(this))

  }
  onHashChange() {
    // localhost:8080/#/home  下面的代码就是拿到 /home
    this.app.current = window.location.hash.slice(1) || '/'
  }
  createRouteMap(options) {
    options.routes.forEach(item => { //这里的item就是自己写的路由配置里面的每一个对象
      this.routeMap[item.path] = item;
    })
  }
  initComponent() {
    //   声明两个全局组件
    Vue.component('router-link', {
      props: {
        to: String
      },
      render(h) {
        // 把这个组件渲染为 <a  href="#/XXX">
        return h('a', {
          attrs: {
            href: '#' + this.to
          }
        }, this.$slots.default) //这是vue写法
        // 也可以使用react也就是es6的jsx语法
        // return (
        // <a href={this.to}>{this.$slots.default}</a>
        // )
      }
    })

    // hash ==> current ==>render
    Vue.component('router-view', {
      // 这里用箭头函数，因为要保留当前this指向VueRouter实例
      render: (h) => {
        // 在这里去拿到组件配置，然后渲染
        const Comp = this.routeMap[this.app.current].component;
        return h(Comp)
      }
    })

  }

}

// 吧VueRouter变成插件,将来Vue.use(VueRouter)的时候就会把Vue本身传递给install方法，等价在这里引入了Vue
VueRouter.install = function (_Vue) {
  Vue = _Vue;    //这里保存了上面就可以使用Vue响应式了

  // 混入
  Vue.mixin({
    beforeCreate() {
      // 这里的代码会混入到Vue执行的时候beforeCreate()执行的时候一起执行。
      // 这样就实现了对Vue的扩展
      // 这里的this指的是vue实例或vue组件实例
      // 这个beforeCreate()里面的代码我们只希望在根组件里面执行一次就可以。
      // 所以就得到为什么要在实例化Vue的时候在options里面加上一个router
      if (this.$options.router) {
        Vue.prototype.router = this.$options.router; //这里的router已经是VueRouter实例并挂载到vue原型上
        this.$options.router.init();

      }
    }
  })

}
export default VueRouter

    # app.vue

    <template>
  <div id="app">
    <router-link to="/">首页</router-link >
    <router-link to="/about">about页</router-link >
    <p>count： {{$store.state.count}}</p>
    <p>score： {{$store.getters.score}}</p>
    <button @click="add">+</button>
    <button @click="addSync">异步 +</button>
   
    <router-view></router-view>
  </div>
</template>

<script>
// import HelloWorld from './components/HelloWorld'
// import Home from './components/Home'
// import About from './components/About'


export default {
  name: 'App',
  components: {
    // HelloWorld
  },
  methods:{
    add(){
      this.$store.commit('add',1)   //做到了响应式，这里修改了state,通知了视图变化
    },
    addSync(){
      this.$store.dispatch('addSync',1)   //做到了响应式，这里修改了state,通知了视图变化
    }
  }
}
</script>
