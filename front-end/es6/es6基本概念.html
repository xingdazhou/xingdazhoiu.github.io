# 1.变量

var 可以重复声明，存在变量提升，无法限制修改（常量），没有块级作用域{}，

let 不能重复声明，是变量可以被修改，块级作用域

const 不能重复声明 ，是常量不能被修改，块级作用域

es5只有函数作用域很全局作用域，es6新增块级作用域

# 2.函数

新增箭头函数

如果只有一个参数，（）可以省略

如果只有一条return，{} 可以省略

# 3.函数的参数

  1. 参数扩展/展开
  2. 默认参数

参数扩展（收集剩余参数，只能放在最后如：...args）

展开数组（展开效果就是直接把数组的内容拿出来）

let arr = [1,2,3]
function show(a,b,c){
    console.log(a,b,c)   //1,2,3
}
show(...arr)

let arr1 = [1,2,3]
let arr2 = [4,5,6]
let arr12 = [...arr1,...arr2]
console.log(arr12)  //1,2,3,4,5,6

//接收剩余参数和数组展开
function show(...args){
    fn(...args)
}
function fn(a,b){
    console.log(a+b)
}
console.log(show(5,5))  //10

    return pre + next

//求平均数
let arr = [1,2,3,4,5,6,7,8,9]
let result = arr.reduce(function(pre,next,index,arr){
    if(index !=arr.length-1){
        return pre + next
    }else{
  		return (pre + next)/arr.length
    }
})
console.log(result)  //5
